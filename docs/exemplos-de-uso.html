Teste Java com RestAssured

import org.junit.jupiter.api.Test;
import static io.restassured.RestAssured.given;

class UsuarioApiTest {
    @Test
    void deveRetornarStatus200AoConsultarUsuarios() {
        given()
            .baseUri("https://api.exemplo.com")
        .when()
            .get("/usuarios")
        .then()
            .statusCode(200);
    }
}
